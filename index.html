// Example API Usage
const response = await fetch('/api/v1/localize', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_API_KEY'
  },
  body: JSON.stringify({
    content: 'Training material text...',
    sourceLanguage: 'en',
    targetLanguages: ['hi', 'bn', 'te'],
    culturalContext: 'automotive_repair'
  })
});

const localizedContent = await response.json();
